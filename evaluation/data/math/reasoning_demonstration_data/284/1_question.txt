Problem: Below is the graph of $y = a \sin (bx + c) + d$ for some positive constants $a,$ $b,$ $c,$ and $d.$  Find $d.$

[asy]import TrigMacros;

size(400);

real f(real x)
{
	return 2*sin(3*x + pi) + 1;
}

draw(graph(f,-3*pi,3*pi,n=700,join=operator ..),red);
trig_axes(-3*pi,3*pi,-4,4,pi/2,1);
layer();
rm_trig_labels(-5,5, 2);

label("$1$", (0,1), E);
label("$2$", (0,2), E);
label("$3$", (0,3), E);
label("$-1$", (0,-1), E);
label("$-2$", (0,-2), E);
label("$-3$", (0,-3), E);
[/asy]

\[ f(x) = a(x-5)^2 + b(x-5) + c \]
\[ g(x) = -a(x+5)^2 - b(x+5) - c \]

\[
(f + g)(x) = a(x-5)^2 + b(x-5) + c - a(x+5)^2 - b(x+5) - c
